الطبقة 1: Client Layer (طبقة العميل)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

فيها تطبيقين:

1. React 18 Web App
   - التطبيق اللي الطلاب والشركات يستخدموه من الكمبيوتر
   - يساعدهم يتصفحوا internships، يقدموا طلبات، يديروا حساباتهم
   - React مختارة لأنها سريعة جداً وسهل نكتب فيها

2. React Native/Expo Mobile App
   - نفس التطبيق بس على الهاتف
   - الفائدة؟ أننا ما نكتب كود مرتين
   - React Native تحول الكود الواحد لـ iOS و Android
   - هذا يوفر وقت وتكاليف كتير


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
الطبقة 2: API Gateway (بوابة البيانات)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

هذي الجسر اللي بينقل الطلبات من التطبيق ل server:

Flask REST API:
   - الـ Server الأساسي اللي يستقبل الطلبات
   - البرنامج يقول: "بدي أتفرج على internships"
   - الـ Server يرد: "ماشي، هجيبها لك دلوقتي"

Blueprint Routing:
   - تنظيم الطلبات - كل حاجة في مكانها
   - مثال: طلبات authentication في مجلد auth/
   - و internships في مجلد internships/
   - الفائدة؟ سهل نلاقي الكود ونصلحه بسرعة

JWT Middleware:
   - لما المستخدم يدخل الحساب، بنديله token
   - ال token دي بتثبت إنك المستخدم الحقيقي
   - كل طلب بعدين بيحمل ال token دي
   - server بيتحقق من البطاقة ويعرف مين أنت بالظبط
   - الفائدة الأمنية؟ لو الـ Server اتقرصن، ال token صعب جداً يحلوها

CORS:
   - دي خاصية أمان في المتصفح
   - بدونها: التطبيق ما يقدر يطلب من server ونقطة
   - معاها: تطبيق الويب والموبايل يقدروا يطلبوا بسلام


━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
الطبقة 3: Business Logic (منطق التطبيق)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

هنا المخ اللي بيفكر ويقرر:

Matching Engine:
   - algorithm ذكية تربط الطالب بأفضل internship
   - تحلل: هذا الطالب عنده مهارات إيه؟
   - وهذا ال internship يطلب مهارات إيه؟
   - ترجع نسبة التطابق (مثلاً 85%)
   - الفائدة؟ الطالب يلاقي وظيفة مناسبة أسرع
   - والشركة تلاقي متقدم جيد فعلاً

CV Builder:
   - أداة تساعد الطالب يعمل سيرة ذاتية احترافية
   - يملأ نموذج: الاسم، الخبرة، المهارات
   - الأداة تحولها لـ PDF جميلة
   - بدل ما الطالب يقضي ساعات يكتب يدوي
   - الفائدة؟ نتيجة احترافية في دقائق

Points System:
   - نظام تحفيز للطلاب
   - يطبق على تدريب = يكسب 10 نقاط
   - يكمل التدريب = يكسب 50 نقطة
   - النقاط يقدر يصرفها على جوائز أو خصومات
   - الفائدة؟ يشجع الطلاب يكملوا ويشتركوا أكتر

AI Chatbot:
   - روبوت ذكي يرد على أسئلة الطلاب
   - الطالب يسأل: "كيف أتقدم؟"
   - الروبوت يجاوب في الحال
   - يعمل 24/7، ما يتعب
   - الفائدة الاقتصادية؟ توفير راتب موظف customer support

الطبقة 4: Data Layer (طبقة البيانات)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

هنا بنخزن كل البيانات:

PostgreSQL على Railway:
   - قاعدة بيانات قوية وآمنة على cloud
   - فيها كل المعلومات: الطلاب، internships، التطبيقات
   - Railway شركة توفر ال cloud مع backup يومي
   - الفائدة؟ البيانات آمنة ومشفرة

SQLAlchemy ORM:
   - هذي library تساعدنا نتعامل مع database
   - بدل ما نكتب SQL معقدة:
     "SELECT * FROM students WHERE age > 20"
   - نكتب Python بسيط:
     "students = Student.query.filter(Student.age > 20).all()"
   - الفائدة؟ 
     1. كود أسهل وأسرع
     2. حماية من هجمات SQL
     3. لو غيرنا database ما نحتاج نعدل الكود

Alembic Migrations:
   - برنامج يتتبع تطور database
   - مثلاً: أضفنا حقل جديد "phone_number" للطالب
   - Alembic يسجل هذا التغيير
   - لو احتجنا نرجع للإصدار القديم = تحذف الحقل تلقائي
   - الفائدة؟ عدم فقدان البيانات عند التحديثات

الطبقة 5: Auth & Security (الأمان والمصادقة)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

حماية التطبيق من المخاطر:

bcrypt Hashing:
   - ما نخزن كلمات المرور نصاً (خطير جداً!)
   - المستخدم يدخل "password123"
   - bcrypt تحولها لـ "$2b$12$x9Hk..." (رقم عشوائي)
   - لو database تسربت = كلمات المرور آمنة
   - الأمان: حتى بـ أقوى حواسيب ما يقدروا يفكوا التشفير

JWT Tokens:
   - بدل ما يرسل username/password على كل طلب (خطر)
   - ينرسل token آمنة
   - server يتحقق من Token بسرعة البرق
   - الفائدة؟ أمان عالي + سرعة عالية

Email Verification:
   - لما ينسجل مستخدم جديد، نتأكد إنه يملك البريد فعلاً
   - يستقبل رسالة تأكيد مع رابط
   - يضغط الرابط = الحساب ينشط
   - الفائدة؟
     1. منع الحسابات الوهمية
     2. نتأكد البريد صحيح عشان نرسل تنبيهات

Role-Based Access:
   - كل نوع مستخدم له صلاحيات مختلفة
   - Student: يشوف internships، يقدم طلبات
   - Company: ينشئ internships، يشوف المتقدمين
   - Admin: يديرها كلها
   - الأمان؟ الطالب ما يقدر يعدل internships حتى لو حاول
